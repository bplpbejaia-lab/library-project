            } catch (error) {
                console.error('Error loading categories:', error);
                document.getElementById('categoriesGrid').innerHTML = '<p class="text-red-600">қ©қъқң л?л? қҰқрл?л?л? қхл?қҰқжл?л?л?қхқҰ</p>';
            }
        }

        // Load all books
        async function loadBooks(search = '', offset = 0) {
            try {
                let url = `/api/books/list?limit=20&offset=${offset}`;
                if (search) {
                    url += '&search=' + encodeURIComponent(search);
                }
                
                const response = await fetch(url);
                const data = await response.json();
                const newBooks = (data.books || []).map(book => ({
                    ...book,
                    img: '/images/book1.png',
                    category: book.category || 'қықхл?'
                }));
                
                // On first load (offset=0), replace books. Otherwise append for pagination
                if (offset === 0) {
                    allBooks = newBooks;
                } else {
                    allBooks = allBooks.concat(newBooks);
                }
                displayBooks(allBooks);
                
                document.getElementById('booksCount').textContent = allBooks.length;
                
                // Show load more button if we got pageSize results (might be more)
                const loadMoreBtn = document.getElementById('loadMoreBtn');
                if (loadMoreBtn) {
                    const pageSize = 20;
                    loadMoreBtn.style.display = (newBooks.length === pageSize) ? 'block' : 'none';
                    if (newBooks.length === pageSize) {
                        loadMoreBtn.onclick = () => loadBooks(search, offset + pageSize);
                    }
                }
            } catch (error) {
                console.error('Error loading books:', error);
                document.getElementById('booksGrid').innerHTML = '<p class="text-red-600">қ©қъқң л?л? қҰқрл?л?л? қхл?л?қҰқщ</p>';
            }
        }

        // Display books
        function displayBooks(books) {
            const booksGrid = document.getElementById('booksGrid');
            if (books.length === 0) {
                booksGrid.innerHTML = '<p class="col-span-full text-center text-slate-500 py-8">л?қх қҰл?қӘқо л?қҰқщ л?қъқхқщл?қё</p>';
                return;
            }
            booksGrid.innerHTML = books.map(book => createBookCard(book)).join('');
        }

        // Filter by category
        function filterByCategory(categoryName) {
            const filtered = allBooks.filter(b => b.category === categoryName);
            displayBooks(filtered);
            document.getElementById('booksCount').textContent = filtered.length;
            window.scrollTo({ top: document.getElementById('booksGrid').offsetTop - 100, behavior: 'smooth' });
        }

        // Search functionality
        document.getElementById('searchInput').addEventListener('input', (e) => {
            const searchTerm = e.target.value.trim();
            if (searchTerm.length === 0) {
                displayBooks(allBooks);
                document.getElementById('booksCount').textContent = allBooks.length;
            } else {
                const filtered = allBooks.filter(b =>
                    b.title.includes(searchTerm) || 
                    (b.author && b.author.includes(searchTerm))
                );
                displayBooks(filtered);
                document.getElementById('booksCount').textContent = filtered.length;
            }
        });

        // Initialize on page load
        document.addEventListener('DOMContentLoaded', () => {
            loadCategories();
            loadBooks();
        });

        // Simple search interactivity
        const searchInput = document.querySelector('input[type="text"]');
        if (searchInput) {
            searchInput.addEventListener('focus', () => {
                searchInput.parentElement.classList.add('ring-4', 'ring-primary/10', 'border-primary');
            });
            searchInput.addEventListener('blur', () => {
                searchInput.parentElement.classList.remove('ring-4', 'ring-primary/10', 'border-primary');
            });
        }
